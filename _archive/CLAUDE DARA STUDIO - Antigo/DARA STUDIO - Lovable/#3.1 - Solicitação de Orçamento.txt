3️⃣ Estrutura técnica (React)
Crie estado global do projeto.

Arquivo sugerido:
src/context/projectContext.ts

Código:
import { createContext, useState } from "react"

export const ProjectContext = createContext(null)

export function ProjectProvider({ children }) {

const [region, setRegion] = useState(null)

return (

<ProjectContext.Provider value={{ region, setRegion }}>
{children}
</ProjectContext.Provider>

)

}

export function getCurrency(region:string){

if(region === "USA"){

return {
symbol: "$",
currency: "USD"
}

}

return {
symbol: "R$",
currency: "BRL"
}

}
