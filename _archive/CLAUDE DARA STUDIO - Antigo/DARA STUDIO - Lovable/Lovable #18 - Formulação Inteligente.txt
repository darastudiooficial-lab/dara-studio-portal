2️⃣ SISTEMA DE ROUGH ESTIMATE AUTOMÁTICO

Criar arquivo:

src/lib/estimateCalculator.ts

Código:

export function calculateEstimate(data:any){

let basePrice = 0

// tipo de projeto
switch(data.projectType){

case "deck":
basePrice = 1200
break

case "garage":
basePrice = 2500
break

case "addition":
basePrice = 4500
break

case "new_home":
basePrice = 7500
break

}

// tamanho

switch(data.size){

case "medium":
basePrice *= 1.3
break

case "large":
basePrice *= 1.6
break

case "xlarge":
basePrice *= 2
break

}

// extras

if(data.deliverables.includes("3d")){
basePrice += 800
}

if(data.deliverables.includes("sections")){
basePrice += 500
}

return basePrice

}
