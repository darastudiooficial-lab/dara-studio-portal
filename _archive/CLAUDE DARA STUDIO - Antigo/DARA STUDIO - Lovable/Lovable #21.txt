7️⃣ Algoritmo de estimate com range

Em vez de número único, usar range.

Arquivo:

estimateCalculator.ts
export function calculateEstimate(data){

let base = 0

if(data.projectType === "deck")
base = 1200

if(data.projectType === "addition")
base = 4500

if(data.projectType === "new_home")
base = 7500

let complexityFactor = 1

if(data.size === "medium")
complexityFactor = 1.3

if(data.size === "large")
complexityFactor = 1.7

const price = base * complexityFactor

return {

min: Math.round(price * 0.9),

max: Math.round(price * 1.25)

}

}
